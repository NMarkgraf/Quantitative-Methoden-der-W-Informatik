\newcommand*{\AUFGABEN}[1]{%
	\mode<beamer>{
	\pause
	\begin{tikzpicture}[remember picture,overlay]
		\node [shift={(-1.8 cm, -1.8 cm)}] at (current page.north east) 
		{%
			\includegraphics[width=3cm]{images/Yellow_Post_It_Note.png}
			\begin{tikzpicture}[remember picture,overlay]
				\node [rotate=-1] at (-1.5 cm, 1.5 cm)
					{%
						\color{green!70!black}%
						\begin{tabular}{c}
							\Large\texttt{Aufgaben:} \\
							\\
							\large\texttt{#1}\\
						\end{tabular}
					};
			\end{tikzpicture}
		};
	\end{tikzpicture}
	}
}
